def compute_windowed_covariance(data, window_size=500, step_size=250):
    """
    Slices the raw continuous timeline into overlapping windows and
    calculates empirical spatial covariance matrices.
    """
    cov_matrices = []
    num_samples = data.shape[0]
    
    for start in range(0, num_samples - window_size, step_size):
        window = data[start:start + window_size, :]
        
        # Center the data window locally (defensive double-centering)
        window_centered = window - np.mean(window, axis=0)
        
        # Compute 4x4 spatial covariance matrix
        cov = (window_centered.T @ window_centered) / (window_size - 1)
        
        # Tikhonov regularization (diagonal loading) to guarantee positive-definiteness
        cov += np.eye(cov.shape[0]) * 1e-6
        cov_matrices.append(cov)
        
    return np.array(cov_matrices)

def riemannian_frechet_mean(covs, max_iter=100, tol=1e-5):
    """
    Computes the true geometric mean (Fréchet Mean) of SPD matrices 
    on the Riemannian manifold, logging actual iterations to convergence.
    """
    n_matrices = len(covs)
    # Initialize with Euclidean mean
    r_mean = np.mean(covs, axis=0)
    
    converged = False
    iterations_run = 0
    
    for i in range(max_iter):
        iterations_run = i + 1
        
        # Matrix square root and inverse square root of current mean
        vals, vecs = eigh(r_mean)
        vals = np.maximum(vals, 1e-12)
        sqrt_mean = vecs @ np.diag(np.sqrt(vals)) @ vecs.T
        inv_sqrt_mean = vecs @ np.diag(1.0 / np.sqrt(vals)) @ vecs.T
        
        # Map matrices to Tangent space
        tangent_vectors = []
        for c in covs:
            proj = inv_sqrt_mean @ c @ inv_sqrt_mean
            tangent_vectors.append(logm(proj))
            
        mean_tangent = np.mean(tangent_vectors, axis=0)
        step = sqrt_mean @ expm(mean_tangent) @ sqrt_mean
        
        diff = np.linalg.norm(step - r_mean, ord='fro')
        r_mean = step
        
        if diff < tol:
            converged = True
            break
            
    return r_mean, iterations_run, converged

def geodesic_distance(A, B):
    """
    Calculates the exact Affine-Invariant Riemannian distance between SPD matrices.
    """
    vals = eigh(B, A, eigvals_only=True)
    return np.sqrt(np.sum(np.log(np.clip(vals, 1e-12, None))**2))

def compute_determinant_swelling(euclidean_mean, riemannian_mean):
    """
    Canonical matrix swelling measurement based on determinant preservation (Arsigny et al. 2007).
    Linear arithmetic averaging of SPD matrices artificially inflates the determinant volume.
    """
    det_e = np.linalg.det(euclidean_mean)
    det_r = np.linalg.det(riemannian_mean)
    return det_e / det_r
